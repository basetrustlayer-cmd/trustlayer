function Table({headers, rows, boldLast}) {
  return (
    <div className="overflow-x-auto mb-4">
      <table className="w-full text-sm border-collapse">
        <thead>
          <tr>{headers.map((h,i)=>(
            <th key={i} className="bg-green-800 text-white px-3 py-2 text-left font-semibold text-xs">{h}</th>
          ))}</tr>
        </thead>
        <tbody>
          {rows.map((row,i)=>(
            <tr key={i} className={boldLast && i===rows.length-1 ? "bg-green-50 font-bold" : i%2===0?"bg-white":"bg-gray-50"}>
              {row.map((cell,j)=>(
                <td key={j} className="px-3 py-2 border-b border-gray-100 text-gray-700">{cell}</td>
              ))}
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}

function Box({color="green", title, children}) {
  const colors = {
    green: "border-green-600 bg-green-50",
    yellow: "border-yellow-500 bg-yellow-50",
    red: "border-red-500 bg-red-50",
    blue: "border-blue-500 bg-blue-50",
    gray: "border-gray-400 bg-gray-50",
  };
  return (
    <div className={`border-l-4 rounded-r-lg p-4 mb-4 ${colors[color]}`}>
      {title && <div className="font-bold text-sm mb-2 text-gray-700">{title}</div>}
      <div className="text-sm text-gray-600">{children}</div>
    </div>
  );
}

function H2({children}) {
  return <div className="text-lg font-bold text-green-800 border-b-2 border-green-700 pb-1 mb-4 mt-2">{children}</div>;
}
function H3({children}) {
  return <div className="font-bold text-gray-700 mb-2 mt-4 text-sm uppercase tracking-wide">{children}</div>;
}
function P({children}) {
  return <p className="text-sm text-gray-600 mb-3 leading-relaxed">{children}</p>;
}

function SectionPage({title, children}) {

    <div className="bg-white rounded-xl shadow-sm border border-gray-200 p-6 mb-4">

  );
}
